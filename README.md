Install nginx service with Ansible

**Playbook file**
root@ip-172-31-11-19:/etc/ansible/playbooks# nano nginx.yaml
---
- name: Configure nginx
  hosts: webservers
  become: yes

  tasks:

    - name: install nginx
      apt: name=nginx update_cache=yes

    - name: restart nginx
      service: name=nginx state=restarted
